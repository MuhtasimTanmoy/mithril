/// F = GF(q)
/// g = F.multiplicative_generator()
/// g = F.multiplicative_generator()
/// assert g.multiplicative_order() == o

/// assert g2.multiplicative_order() == 2**s

///     return [int(int(val) // (2 ** (width * i)) % 2 ** width) for i in range(n)]
